            }
            
            // 
            // –ú–ï–¢–û–î –ü–ï–†–ï–ö–õ–Æ–ß–ï–ù–ò–Ø –ú–ï–ñ–î–£ –≠–ö–†–ê–ù–ê–ú–ò
            // –ü–æ–∫–∞–∑—ã–≤–∞–µ—Ç —É–∫–∞–∑–∞–Ω–Ω—ã–π —ç–∫—Ä–∞–Ω –∏ —Å–∫—Ä—ã–≤–∞–µ—Ç –æ—Å—Ç–∞–ª—å–Ω—ã–µ

            showScreen(screenId) {

            // –û—Ç–æ–±—Ä–∞–∂–∞–µ—Ç –∞–∫—Ç—É–∞–ª—å–Ω—É—é —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫—É –Ω–∞ –≥–ª–∞–≤–Ω–æ–º —ç–∫—Ä–∞–Ω–µ

            // –ü–æ–¥–≥–æ—Ç–∞–≤–ª–∏–≤–∞–µ—Ç –¥–∞–Ω–Ω—ã–µ –¥–ª—è –Ω–æ–≤–æ–≥–æ —É—Ä–æ–≤–Ω—è
            //
            startGame() {
                console.log('üéØ –ó–∞–ø—É—Å–∫ –∏–≥—Ä—ã...');  // –û—Ç–ª–∞–¥–æ—á–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ
                
                // –°–ë–†–û–° –í–†–ï–ú–ï–ù–ù–´–• –î–ê–ù–ù–´–• –î–õ–Ø –ù–û–í–û–ì–û –£–†–û–í–ù–Ø
                this.state.currentInput = [];  // –û—á–∏—â–∞–µ–º –Ω–∞–±—Ä–∞–Ω–Ω—ã–µ –±—É–∫–≤—ã
                this.state.foundWords = [];    // –û—á–∏—â–∞–µ–º —Å–ø–∏—Å–æ–∫ –Ω–∞–π–¥–µ–Ω–Ω—ã—Ö —Å–ª–æ–≤

                // –û–ë–ù–û–í–õ–ï–ù–ò–ï –ò–ù–¢–ï–†–§–ï–ô–°–ê –ò–ì–†–û–í–û–ì–û –≠–ö–†–ê–ù–ê

            // –ú–ï–¢–û–î –ì–ï–ù–ï–†–ê–¶–ò–ò –î–ê–ù–ù–´–• –£–†–û–í–ù–Ø
            // –°–æ–∑–¥–∞–µ—Ç —Å–ª—É—á–∞–π–Ω—ã–π –Ω–∞–±–æ—Ä —Å–ª–æ–≤ –∏ –±—É–∫–≤ –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ —É—Ä–æ–≤–Ω—è
            //
            generateLevel() {
                console.log('üé≤ –ì–µ–Ω–µ—Ä–∞—Ü–∏—è —É—Ä–æ–≤–Ω—è', this.state.level);  // –û—Ç–ª–∞–¥–æ—á–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ
                
                // 
                // –û–ü–†–ï–î–ï–õ–ï–ù–ò–ï –ö–û–õ–ò–ß–ï–°–¢–í–ê –°–õ–û–í –ù–ê –£–†–û–í–ù–ï
                // –ß–µ–º –≤—ã—à–µ —É—Ä–æ–≤–µ–Ω—å, —Ç–µ–º –±–æ–ª—å—à–µ —Å–ª–æ–≤ –Ω—É–∂–Ω–æ –Ω–∞–π—Ç–∏
                // –ú–∏–Ω–∏–º—É–º 3 —Å–ª–æ–≤–∞, –º–∞–∫—Å–∏–º—É–º 5 —Å–ª–æ–≤
                // –ö–∞–∂–¥—ã–µ 2 —É—Ä–æ–≤–Ω—è –¥–æ–±–∞–≤–ª—è–µ–º –ø–æ –æ–¥–Ω–æ–º—É —Å–ª–æ–≤—É
                //
                const wordCount = Math.min(3 + Math.floor(this.state.level / 2), 5);
                
                // 
                // –í–´–ë–û–† –°–õ–£–ß–ê–ô–ù–´–• –°–õ–û–í –ò–ó –°–ü–ò–°–ö–ê
                //

                // –°–æ–∑–¥–∞–µ–º –∫–æ–ø–∏—é —Å–ø–∏—Å–∫–∞ —Å–ª–æ–≤, —á—Ç–æ–±—ã –Ω–µ –∏–∑–º–µ–Ω—è—Ç—å –æ—Ä–∏–≥–∏–Ω–∞–ª
                let availableWords = [...this.wordList];
                
                // –ú–∞—Å—Å–∏–≤ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤—ã–±—Ä–∞–Ω–Ω—ã—Ö —Å–ª–æ–≤
                let selectedWords = [];

                for (let i = 0; i < wordCount; i++) {
                    // –ï—Å–ª–∏ —Å–ª–æ–≤–∞ –∑–∞–∫–æ–Ω—á–∏–ª–∏—Å—å, –ø—Ä–µ—Ä—ã–≤–∞–µ–º —Ü–∏–∫–ª
                    if (availableWords.length === 0) break;
                    
                    // –í—ã–±–∏—Ä–∞–µ–º —Å–ª—É—á–∞–π–Ω—ã–π –∏–Ω–¥–µ–∫—Å
                    const randomIndex = Math.floor(Math.random() * availableWords.length);
                    
                    // –ü–æ–ª—É—á–∞–µ–º —Å–ª–æ–≤–æ –ø–æ —Å–ª—É—á–∞–π–Ω–æ–º—É –∏–Ω–¥–µ–∫—Å—É
                    const word = availableWords[randomIndex];
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º —Å–ª–æ–≤–æ –≤ –≤—ã–±—Ä–∞–Ω–Ω—ã–µ
                    selectedWords.push(word);
                    
                    // –£–¥–∞–ª—è–µ–º —Å–ª–æ–≤–æ –∏–∑ –¥–æ—Å—Ç—É–ø–Ω—ã—Ö, —á—Ç–æ–±—ã –Ω–µ –≤—ã–±–∏—Ä–∞—Ç—å –µ–≥–æ —Å–Ω–æ–≤–∞
                    availableWords.splice(randomIndex, 1);
                }

                // –°–û–ë–ò–†–ê–ï–ú –í–°–ï –ë–£–ö–í–´ –ò–ó –í–´–ë–†–ê–ù–ù–´–• –°–õ–û–í
                // –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Å–ª–æ–≤–∞ –≤ –º–∞—Å—Å–∏–≤—ã –±—É–∫–≤ –∏ –æ–±—ä–µ–¥–∏–Ω—è–µ–º –∏—Ö
                //
                let allLetters = [];
                selectedWords.forEach(word => {
                    // split('') —Ä–∞–∑–±–∏–≤–∞–µ—Ç —Å—Ç—Ä–æ–∫—É –Ω–∞ –º–∞—Å—Å–∏–≤ –±—É–∫–≤
                    // ... (spread operator) –¥–æ–±–∞–≤–ª—è–µ—Ç –≤—Å–µ –±—É–∫–≤—ã –≤ allLetters
                    allLetters.push(...word.split(''));
                });
                
                // 
                // –í–´–ë–ò–†–ê–ï–ú –£–ù–ò–ö–ê–õ–¨–ù–´–ï –ë–£–ö–í–´ –î–õ–Ø –ö–†–£–ì–ê
                // Set –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —É–¥–∞–ª—è–µ—Ç –¥—É–±–ª–∏–∫–∞—Ç—ã
                //
                const uniqueLetters = [...new Set(allLetters)];
                let availableLetters = uniqueLetters;
                
                // 
                // –ï–°–õ–ò –ë–£–ö–í –ü–û–õ–£–ß–ò–õ–û–°–¨ –ú–ê–õ–û, –î–û–ë–ê–í–õ–Ø–ï–ú –°–õ–£–ß–ê–ô–ù–´–ï
                // –ö—Ä—É–≥ –¥–æ–ª–∂–µ–Ω —Å–æ–¥–µ—Ä–∂–∞—Ç—å –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –±—É–∫–≤ –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –∏–≥—Ä—ã
                //
                if (availableLetters.length < 6) {
                    // –†—É—Å—Å–∫–∏–π –∞–ª—Ñ–∞–≤–∏—Ç –¥–ª—è –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —Å–ª—É—á–∞–π–Ω—ã—Ö –±—É–∫–≤
                    const alphabet = '–ê–ë–í–ì–î–ï–Å–ñ–ó–ò–ô–ö–õ–ú–ù–û–ü–†–°–¢–£–§–•–¶–ß–®–©–™–´–¨–≠–Æ–Ø';
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º –±—É–∫–≤—ã –ø–æ–∫–∞ –Ω–µ –Ω–∞–±–µ—Ä–µ–º –º–∏–Ω–∏–º—É–º 8
                    while (availableLetters.length < 8) {
                        // –í—ã–±–∏—Ä–∞–µ–º —Å–ª—É—á–∞–π–Ω—É—é –±—É–∫–≤—É –∏–∑ –∞–ª—Ñ–∞–≤–∏—Ç–∞
                        const randomLetter = alphabet[Math.floor(Math.random() * alphabet.length)];
                        
                        // –ï—Å–ª–∏ —Ç–∞–∫–æ–π –±—É–∫–≤—ã –µ—â–µ –Ω–µ—Ç –≤ —Å–ø–∏—Å–∫–µ, –¥–æ–±–∞–≤–ª—è–µ–º
                        if (!availableLetters.includes(randomLetter)) {
                            availableLetters.push(randomLetter);
                        }
                    }
                }
                
                // 
                // –û–ì–†–ê–ù–ò–ß–ò–í–ê–ï–ú –ö–û–õ–ò–ß–ï–°–¢–í–û –ë–£–ö–í –î–û 10
                // –ß—Ç–æ–±—ã –∫—Ä—É–≥ –Ω–µ –±—ã–ª –ø–µ—Ä–µ–≥—Ä—É–∂–µ–Ω
                //
                availableLetters = availableLetters.slice(0, 10);
                
                // 
                // –ü–ï–†–ï–ú–ï–®–ò–í–ê–ï–ú –ë–£–ö–í–´ –î–õ–Ø –†–ê–ó–ù–û–û–ë–†–ê–ó–ò–Ø
                // –ê–ª–≥–æ—Ä–∏—Ç–º –§–∏—à–µ—Ä–∞-–ô–µ—Ç—Å–∞ (Fisher-Yates shuffle)
                //
                for (let i = availableLetters.length - 1; i > 0; i--) {
                    // –í—ã–±–∏—Ä–∞–µ–º —Å–ª—É—á–∞–π–Ω—ã–π –∏–Ω–¥–µ–∫—Å –æ—Ç 0 –¥–æ i
                    const j = Math.floor(Math.random() * (i + 1));
                    
                    // –ú–µ–Ω—è–µ–º –º–µ—Å—Ç–∞–º–∏ —ç–ª–µ–º–µ–Ω—Ç—ã i –∏ j
                    [availableLetters[i], availableLetters[j]] = [availableLetters[j], availableLetters[i]];
                }
                
                // 
                // –°–û–ó–î–ê–ï–ú –°–¢–†–£–ö–¢–£–†–£ –î–õ–Ø –û–¢–û–ë–†–ê–ñ–ï–ù–ò–Ø –°–õ–û–í
                // –î–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–ª–æ–≤–∞ —Å–æ–∑–¥–∞–µ–º –º–∞—Å—Å–∏–≤ –±—É–∫–≤ —Å —Å–æ—Å—Ç–æ—è–Ω–∏–µ–º (–æ—Ç–∫—Ä—ã—Ç–∞/—Å–∫—Ä—ã—Ç–∞)
                //
                const wordCells = {};
                selectedWords.forEach(word => {
                    // –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Å–ª–æ–≤–æ –≤ –º–∞—Å—Å–∏–≤ –æ–±—ä–µ–∫—Ç–æ–≤
                    wordCells[word] = word.split('').map((letter, index) => ({
                        revealed: false,  // –ü–æ —É–º–æ–ª—á–∞–Ω–∏—é –±—É–∫–≤–∞ —Å–∫—Ä—ã—Ç–∞
                        letter: letter,   // –°–∞–º–∞ –±—É–∫–≤–∞
                        index: index      // –ü–æ–∑–∏—Ü–∏—è –±—É–∫–≤—ã –≤ —Å–ª–æ–≤–µ
                    }));
                });
                
                // 
                // –°–û–•–†–ê–ù–Ø–ï–ú –í–°–ï –î–ê–ù–ù–´–ï –£–†–û–í–ù–Ø –í –°–û–°–¢–û–Ø–ù–ò–ò –ò–ì–†–´
                //
                this.state.levelData = {
                    availableLetters: availableLetters,  // –ë—É–∫–≤—ã –Ω–∞ –∫—Ä—É–≥–µ
                    targetWords: selectedWords,          // –°–ª–æ–≤–∞, –∫–æ—Ç–æ—Ä—ã–µ –Ω—É–∂–Ω–æ –Ω–∞–π—Ç–∏
                    wordCells: wordCells,                // –°—Ç—Ä—É–∫—Ç—É—Ä–∞ –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è —Å–ª–æ–≤
                    allLetters: allLetters               // –í—Å–µ –±—É–∫–≤—ã –∏–∑ –≤—ã–±—Ä–∞–Ω–Ω—ã—Ö —Å–ª–æ–≤
                };
                
                console.log('‚úÖ –£—Ä–æ–≤–µ–Ω—å —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω:', selectedWords);  // –û—Ç–ª–∞–¥–æ—á–Ω–æ–µ —Å–æ–æ–±—â–µ–Ω–∏–µ
            }
            
            // 
            // –ú–ï–¢–û–î –û–ë–ù–û–í–õ–ï–ù–ò–Ø –ò–ù–¢–ï–†–§–ï–ô–°–ê –ò–ì–†–û–í–û–ì–û –≠–ö–†–ê–ù–ê
            // –ü–µ—Ä–µ—Ä–∏—Å–æ–≤—ã–≤–∞–µ—Ç –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã –∏–≥—Ä–æ–≤–æ–≥–æ —ç–∫—Ä–∞–Ω–∞
            //
            updateGameScreen() {
                // 
                // –û–ë–ù–û–í–õ–ï–ù–ò–ï –°–¢–ê–¢–ò–°–¢–ò–ö–ò –í –•–ï–î–ï–†–ï
                //

                // –¢–µ–∫—É—â–∏–π —É—Ä–æ–≤–µ–Ω—å
                document.getElementById('currentLevel').textContent = this.state.level;
                
                // –¢–µ–∫—É—â–∏–µ –æ—á–∫–∏
                document.getElementById('score').textContent = this.state.score;
                
                // 
                // –û–¢–û–ë–†–ê–ñ–ï–ù–ò–ï –°–ï–¢–ö–ò –°–û –°–õ–û–í–ê–ú–ò
                //
                this.renderWordsGrid();
                
                // 
                // –û–¢–û–ë–†–ê–ñ–ï–ù–ò–ï –û–ë–õ–ê–°–¢–ò –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
                //
                this.renderCurrentInput();
                
                // 
                // –û–¢–û–ë–†–ê–ñ–ï–ù–ò–ï –ö–†–£–ì–ê –° –ë–£–ö–í–ê–ú–ò
                //
                this.renderCircleInput();
                
                // 
                // –û–ë–ù–û–í–õ–ï–ù–ò–ï –ò–ù–§–û–†–ú–ê–¶–ò–û–ù–ù–û–ô –°–¢–†–û–ö–ò
                // –ü–æ–∫–∞–∑—ã–≤–∞–µ—Ç –ø—Ä–æ–≥—Ä–µ—Å—Å –ø–æ–∏—Å–∫–∞ —Å–ª–æ–≤
                //
                const foundCount = this.state.foundWords.length;
                const totalWords = this.state.levelData.targetWords.length;
                document.getElementById('wordInfo').textContent = 
                    `–ù–∞–π–¥–µ–Ω–æ —Å–ª–æ–≤: ${foundCount}/${totalWords}`;
            }
            
            // 
            // –ú–ï–¢–û–î –û–¢–û–ë–†–ê–ñ–ï–ù–ò–Ø –°–ï–¢–ö–ò –°–û –°–õ–û–í–ê–ú–ò
            // –°–æ–∑–¥–∞–µ—Ç –≤–∏–∑—É–∞–ª—å–Ω–æ–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω–∏–µ —Å–ª–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –Ω—É–∂–Ω–æ –Ω–∞–π—Ç–∏
            //
            renderWordsGrid() {
                // –ü–æ–ª—É—á–∞–µ–º –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä –¥–ª—è —Å–µ—Ç–∫–∏ —Å–ª–æ–≤
                const container = document.getElementById('wordsGrid');
                
                // –û—á–∏—â–∞–µ–º –ø—Ä–µ–¥—ã–¥—É—â–µ–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
                container.innerHTML = '';
                
                // –î–ª—è –∫–∞–∂–¥–æ–≥–æ —Å–ª–æ–≤–∞ –≤ —É—Ä–æ–≤–Ω–µ
                this.state.levelData.targetWords.forEach(word => {
                    // –°–æ–∑–¥–∞–µ–º –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä –¥–ª—è –æ–¥–Ω–æ–≥–æ —Å–ª–æ–≤–∞ (—Ä—è–¥)
                    const wordRow = document.createElement('div');
                    wordRow.className = 'word-row';
                    
                    // –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω–∞–π–¥–µ–Ω–æ –ª–∏ —ç—Ç–æ —Å–ª–æ–≤–æ
                    const isFound = this.state.foundWords.includes(word);
                    
                    // –î–ª—è –∫–∞–∂–¥–æ–π –±—É–∫–≤—ã –≤ —Å–ª–æ–≤–µ
                    this.state.levelData.wordCells[word].forEach((cell, index) => {
                        // –°–æ–∑–¥–∞–µ–º —ç–ª–µ–º–µ–Ω—Ç –¥–ª—è –±—É–∫–≤—ã
                        const cellElement = document.createElement('div');
                        cellElement.className = 'letter-cell';
                        
                        // –ï—Å–ª–∏ —Å–ª–æ–≤–æ –Ω–∞–π–¥–µ–Ω–æ –ò–õ–ò –±—É–∫–≤–∞ –æ—Ç–∫—Ä—ã—Ç–∞ (–ø–æ–¥—Å–∫–∞–∑–∫–æ–π)
                        if (isFound || cell.revealed) {
                            // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –±—É–∫–≤—É
                            cellElement.textContent = cell.letter;
                            cellElement.classList.add('revealed');
                        }
                        // –ò–Ω–∞—á–µ –±—É–∫–≤–∞ –æ—Å—Ç–∞–µ—Ç—Å—è —Å–∫—Ä—ã—Ç–æ–π (–æ—Ç–æ–±—Ä–∞–∂–∞–µ—Ç—Å—è –∫–∞–∫ –ø—É—Å—Ç–æ–π –∫–≤–∞–¥—Ä–∞—Ç)
                        
                        // –î–æ–±–∞–≤–ª—è–µ–º –±—É–∫–≤—É –≤ —Ä—è–¥
                        wordRow.appendChild(cellElement);
                    });
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º —Ä—è–¥ –≤ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä
                    container.appendChild(wordRow);
                });
            }
            
            // 
            // –ú–ï–¢–û–î –û–¢–û–ë–†–ê–ñ–ï–ù–ò–Ø –û–ë–õ–ê–°–¢–ò –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
            // –ü–æ–∫–∞–∑—ã–≤–∞–µ—Ç –±—É–∫–≤—ã, –∫–æ—Ç–æ—Ä—ã–µ –∏–≥—Ä–æ–∫ –≤—ã–±—Ä–∞–ª –∏–∑ –∫—Ä—É–≥–∞
            //
            renderCurrentInput() {
                // –ü–æ–ª—É—á–∞–µ–º –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä –¥–ª—è —Ç–µ–∫—É—â–µ–≥–æ –≤–≤–æ–¥–∞
                const container = document.getElementById('currentInput');
                
                // –û—á–∏—â–∞–µ–º –ø—Ä–µ–¥—ã–¥—É—â–µ–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
                container.innerHTML = '';
                
                // –î–ª—è –∫–∞–∂–¥–æ–π –±—É–∫–≤—ã –≤ —Ç–µ–∫—É—â–µ–º –≤–≤–æ–¥–µ
                this.state.currentInput.forEach(letter => {
                    // –°–æ–∑–¥–∞–µ–º —ç–ª–µ–º–µ–Ω—Ç –¥–ª—è –±—É–∫–≤—ã
                    const div = document.createElement('div');
                    div.className = 'input-letter';
                    div.textContent = letter;
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º –±—É–∫–≤—É –≤ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä
                    container.appendChild(div);
                });
            }
            
            // 
            // –ú–ï–¢–û–î –û–¢–û–ë–†–ê–ñ–ï–ù–ò–Ø –ö–†–£–ì–ê –° –ë–£–ö–í–ê–ú–ò
            // –°–æ–∑–¥–∞–µ—Ç –∏–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω—ã–π –∫—Ä—É–≥ —Å –±—É–∫–≤–∞–º–∏
            //
            renderCircleInput() {
                // –ü–æ–ª—É—á–∞–µ–º –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä –¥–ª—è –∫—Ä—É–≥–∞
                const container = document.getElementById('circleInput');
                
                // –û—á–∏—â–∞–µ–º –ø—Ä–µ–¥—ã–¥—É—â–µ–µ —Å–æ–¥–µ—Ä–∂–∏–º–æ–µ
                container.innerHTML = '';
                
                // 
                // –°–û–ó–î–ê–ï–ú –¶–ï–ù–¢–†–ê–õ–¨–ù–£–Æ –ö–ù–û–ü–ö–£ –ü–û–î–¢–í–ï–†–ñ–î–ï–ù–ò–Ø
                //
                const center = document.createElement('div');
                center.className = 'circle-center';
                center.innerHTML = '<i class="fas fa-check"></i>';  // –ò–∫–æ–Ω–∫–∞ –≥–∞–ª–æ—á–∫–∏
                center.onclick = () => this.submitWord();  // –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–ª–∏–∫–∞
                container.appendChild(center);
                
                // 
                // –°–û–ó–î–ê–ï–ú –ë–£–ö–í–´ –ü–û –ö–†–£–ì–£
                //
                const letters = this.state.levelData.availableLetters;
                const radius = 85;       // –†–∞–¥–∏—É—Å –∫—Ä—É–≥–∞ –¥–ª—è —Ä–∞–∑–º–µ—â–µ–Ω–∏—è –±—É–∫–≤
                const centerX = 125;     // X-–∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ü–µ–Ω—Ç—Ä–∞ –∫—Ä—É–≥–∞
                const centerY = 125;     // Y-–∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞ —Ü–µ–Ω—Ç—Ä–∞ –∫—Ä—É–≥–∞
                
                // –î–ª—è –∫–∞–∂–¥–æ–π –±—É–∫–≤—ã –≤ —Å–ø–∏—Å–∫–µ –¥–æ—Å—Ç—É–ø–Ω—ã—Ö –±—É–∫–≤
                letters.forEach((letter, index) => {
                    // 
                    // –í–´–ß–ò–°–õ–Ø–ï–ú –ü–û–ó–ò–¶–ò–Æ –ë–£–ö–í–´ –ù–ê –û–ö–†–£–ñ–ù–û–°–¢–ò
                    //

                    // –í—ã—á–∏—Å–ª—è–µ–º —É–≥–æ–ª –¥–ª—è —ç—Ç–æ–π –±—É–∫–≤—ã (–≤ —Ä–∞–¥–∏–∞–Ω–∞—Ö)
                    // 2 * Math.PI = 360 –≥—Ä–∞–¥—É—Å–æ–≤ –≤ —Ä–∞–¥–∏–∞–Ω–∞—Ö
                    const angle = (index / letters.length) * 2 * Math.PI;
                    
                    // –í—ã—á–∏—Å–ª—è–µ–º –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã X –∏ Y –Ω–∞ –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏
                    const x = centerX + radius * Math.cos(angle);
                    const y = centerY + radius * Math.sin(angle);
                    
                    // 
                    // –°–û–ó–î–ê–ï–ú –≠–õ–ï–ú–ï–ù–¢ –î–õ–Ø –ë–£–ö–í–´
                    //
                    const btn = document.createElement('div');
                    btn.className = 'circle-letter';
                    btn.textContent = letter;
                    
                    // –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –ø–æ–∑–∏—Ü–∏—é (—Å–º–µ—â–∞–µ–º –Ω–∞ –ø–æ–ª–æ–≤–∏–Ω—É —Ä–∞–∑–º–µ—Ä–∞ –¥–ª—è —Ü–µ–Ω—Ç—Ä–∏—Ä–æ–≤–∞–Ω–∏—è)
                    btn.style.left = `${x - 20}px`;  // 20 = –ø–æ–ª–æ–≤–∏–Ω–∞ —à–∏—Ä–∏–Ω—ã (40px)
                    btn.style.top = `${y - 20}px`;   // 20 = –ø–æ–ª–æ–≤–∏–Ω–∞ –≤—ã—Å–æ—Ç—ã (40px)
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º –æ–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–ª–∏–∫–∞
                    btn.onclick = () => this.addLetter(letter);
                    
                    // –î–æ–±–∞–≤–ª—è–µ–º –±—É–∫–≤—É –≤ –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä
                    container.appendChild(btn);
                });
            }
            
            // 
            // –ú–ï–¢–û–î –î–û–ë–ê–í–õ–ï–ù–ò–Ø –ë–£–ö–í–´ –í –¢–ï–ö–£–©–ò–ô –í–í–û–î
            // –í—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–∏ –∫–ª–∏–∫–µ –Ω–∞ –±—É–∫–≤—É –≤ –∫—Ä—É–≥–µ
            //
            addLetter(letter) {
                // 
                // –ü–†–û–í–ï–†–ö–ê: –ú–û–ñ–ï–ú –õ–ò –ú–´ –ò–°–ü–û–õ–¨–ó–û–í–ê–¢–¨ –≠–¢–£ –ë–£–ö–í–£ –ï–©–ï –†–ê–ó
                //

                // –°—á–∏—Ç–∞–µ–º, —Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ —ç—Ç–∞ –±—É–∫–≤–∞ —É–∂–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∞ –≤ —Ç–µ–∫—É—â–µ–º –≤–≤–æ–¥–µ
                const usedCount = this.state.currentInput.filter(l => l === letter).length;
                
                // –°—á–∏—Ç–∞–µ–º, —Å–∫–æ–ª—å–∫–æ —Ä–∞–∑ —ç—Ç–∞ –±—É–∫–≤–∞ –¥–æ—Å—Ç—É–ø–Ω–∞ –≤ —É—Ä–æ–≤–Ω–µ
                const availableCount = this.state.levelData.allLetters.filter(l => l === letter).length;
                
                // –ï—Å–ª–∏ –º—ã –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–ª–∏ –±—É–∫–≤—É –º–µ–Ω—å—à–µ —Ä–∞–∑, —á–µ–º –æ–Ω–∞ –¥–æ—Å—Ç—É–ø–Ω–∞
                if (usedCount < availableCount) {
                    // –î–æ–±–∞–≤–ª—è–µ–º –±—É–∫–≤—É –≤ —Ç–µ–∫—É—â–∏–π –≤–≤–æ–¥
                    this.state.currentInput.push(letter);
                    
                    // –û–±–Ω–æ–≤–ª—è–µ–º –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å
                    this.updateGameScreen();
                }
                // –ò–Ω–∞—á–µ –±—É–∫–≤–∞ –Ω–µ–¥–æ—Å—Ç—É–ø–Ω–∞ (–≤—Å–µ —ç–∫–∑–µ–º–ø–ª—è—Ä—ã —É–∂–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω—ã)
            }
            
            // 
            // –ú–ï–¢–û–î –£–î–ê–õ–ï–ù–ò–Ø –ü–û–°–õ–ï–î–ù–ï–ô –ë–£–ö–í–´ –ò–ó –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
            // –í—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø–æ Backspace –∏–ª–∏ –∫–Ω–æ–ø–∫–µ –æ—á–∏—Å—Ç–∫–∏
            //
            removeLastLetter() {
                // –ï—Å–ª–∏ –≤ —Ç–µ–∫—É—â–µ–º –≤–≤–æ–¥–µ –µ—Å—Ç—å —Ö–æ—Ç—è –±—ã –æ–¥–Ω–∞ –±—É–∫–≤–∞
                if (this.state.currentInput.length > 0) {
                    // –£–¥–∞–ª—è–µ–º –ø–æ—Å–ª–µ–¥–Ω—é—é –±—É–∫–≤—É (–º–µ—Ç–æ–¥ pop —É–¥–∞–ª—è–µ—Ç –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–æ—Å–ª–µ–¥–Ω–∏–π —ç–ª–µ–º–µ–Ω—Ç)
                    this.state.currentInput.pop();
                    
                    // –û–±–Ω–æ–≤–ª—è–µ–º –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å
                    this.updateGameScreen();
                }
            }
            
            // 
            // –ú–ï–¢–û–î –û–ß–ò–°–¢–ö–ò –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
            // –£–¥–∞–ª—è–µ—Ç –≤—Å–µ –Ω–∞–±—Ä–∞–Ω–Ω—ã–µ –±—É–∫–≤—ã
            //
            clearInput() {
                // –û—á–∏—â–∞–µ–º –º–∞—Å—Å–∏–≤ —Ç–µ–∫—É—â–µ–≥–æ –≤–≤–æ–¥–∞
                this.state.currentInput = [];
                
                // –û–±–Ω–æ–≤–ª—è–µ–º –∏–Ω—Ç–µ—Ä—Ñ–µ–π—Å
                this.updateGameScreen();
            }
            
            // 
            // –ú–ï–¢–û–î –°–ë–†–û–°–ê –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
            // –í –¥–∞–Ω–Ω–æ–π —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏ –∞–Ω–∞–ª–æ–≥–∏—á–µ–Ω clearInput
            //
            resetInput() {
                this.clearInput();
            }
            
            // 
            // –ú–ï–¢–û–î –ü–†–û–í–ï–†–ö–ò –ò –û–¢–ü–†–ê–í–ö–ò –°–õ–û–í–ê
            // –í—ã–∑—ã–≤–∞–µ—Ç—Å—è –ø—Ä–∏ –∫–ª–∏–∫–µ –Ω–∞ —Ü–µ–Ω—Ç—Ä–∞–ª—å–Ω—É—é –∫–Ω–æ–ø–∫—É –∏–ª–∏ –Ω–∞–∂–∞—Ç–∏–∏ Enter
            //
            submitWord() {
                // 
                // –§–û–†–ú–ò–†–£–ï–ú –°–õ–û–í–û –ò–ó –¢–ï–ö–£–©–ï–ì–û –í–í–û–î–ê
                // join('') –æ–±—ä–µ–¥–∏–Ω—è–µ—Ç –º–∞—Å—Å–∏–≤ –±—É–∫–≤ –≤ —Å—Ç—Ä–æ–∫—É
                // toUpperCase() –ø—Ä–∏–≤–æ–¥–∏—Ç –∫ –≤–µ—Ä—Ö–Ω–µ–º—É —Ä–µ–≥–∏—Å—Ç—Ä—É (–¥–ª—è –Ω–∞–¥–µ–∂–Ω–æ—Å—Ç–∏)
                //
                const word = this.state.currentInput.join('').toUpperCase();
                
                // 
                // –ü–†–û–í–ï–†–ö–ê: –°–õ–û–í–û –î–û–õ–ñ–ù–û –ë–´–¢–¨ –•–û–¢–Ø –ë–´ –ò–ó 2 –ë–£–ö–í
